server {
    listen 80;
    listen [::]:80;
    server_name backyard-tech.ru www.backyard-tech.ru;

    root /var/www/backyard-tech/dist;
    index index.html;

    # Real IP from Cloudflare (нужно будет расширить CF IP list в Session 2)
    set_real_ip_from 0.0.0.0/0;
    real_ip_header CF-Connecting-IP;

    # Redirect www → apex
    if ($host = www.backyard-tech.ru) {
        return 301 http://backyard-tech.ru$request_uri;
    }

    # Gzip
    gzip on;
    gzip_vary on;
    gzip_types text/plain text/css application/javascript application/json image/svg+xml application/xml+rss;
    gzip_min_length 1000;

    # Cache headers для статики
    location /static/ {
        expires 7d;
        add_header Cache-Control "public, immutable";
    }

    # HTML не кешируем
    location ~* \.html$ {
        add_header Cache-Control "no-cache";
    }

    # Основной handler
    location / {
        try_files $uri $uri/ =404;
    }

    # Security headers (базовые)
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;
}
