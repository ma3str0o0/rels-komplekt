===== unit: rels-notify.service =====
[Unit]
Description=Рельс-Комплект API (gunicorn)
After=network.target

[Service]
Type=simple
User=rels-app
Group=rels-app
WorkingDirectory=/root/projects/rels-komplekt
RuntimeDirectory=rels-notify
RuntimeDirectoryMode=0750
ExecStart=/usr/local/bin/gunicorn \
    --bind unix:/run/rels-notify/app.sock \
    --workers 2 \
    --timeout 60 \
    --access-logfile - \
    --error-logfile - \
    notify_app:application
Restart=always
RestartSec=3
StandardOutput=journal
StandardError=journal
SyslogIdentifier=rels-notify

[Install]
WantedBy=multi-user.target
