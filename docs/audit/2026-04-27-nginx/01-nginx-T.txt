# Rate limit: не более 10 заявок/мин с одного IP (notify)
limit_req_zone $binary_remote_addr zone=api_notify:10m rate=10r/m;
# Rate limit: не более 120 событий/мин с одного IP (tracker)
limit_req_zone $binary_remote_addr zone=api_track:10m  rate=120r/m;
# Rate limit: не более 3 заявок/мин с одного IP (lead CRM)
limit_req_zone $binary_remote_addr zone=api_lead:10m   rate=3r/m;

server {
    listen 8080;
    server_name _;
    root /root/projects/rels-komplekt;
    charset utf-8;
    client_max_body_size 11m;

    # ── Блокируем служебные пути ──────────────────────────────────
    location ~ ^/(proxy|\.git|data/vsp74_scrape|data/catalog_enriched|data/catalog_backup|data/catalog\.json\.bak|data/image_map|data/competitor|data/eshop|data/metrics|docs|tools|\.env|CLAUDE\.md|DEVLOG\.md|PROJECT_STATUS\.md) {
        return 403;
    }

    # ── Трекер аналитики → gunicorn ──────────────────────────────
    location /api/track {
        limit_req zone=api_track burst=20 nodelay;
        limit_req_status 429;
        proxy_pass          http://unix:/run/rels-notify/app.sock;
        proxy_set_header    Host $host;
        proxy_set_header    X-Real-IP $remote_addr;
        proxy_read_timeout  10s;
        proxy_connect_timeout 5s;
        proxy_send_timeout  10s;
        include /etc/nginx/snippets/security-headers.conf;
    }

    # ── CRM-заявки → gunicorn ────────────────────────────────────
    location /api/lead {
        limit_req zone=api_lead burst=2 nodelay;
        limit_req_status 429;
        proxy_pass          http://unix:/run/rels-notify/app.sock;
        proxy_set_header    Host $host;
        proxy_set_header    X-Real-IP $remote_addr;
        proxy_read_timeout  15s;
        proxy_connect_timeout 5s;
        proxy_send_timeout  15s;
        include /etc/nginx/snippets/security-headers.conf;
    }

    # ── API → gunicorn с rate limiting ───────────────────────────
    location /api/notify {
        limit_req zone=api_notify burst=3 nodelay;
        limit_req_status 429;
        proxy_pass          http://unix:/run/rels-notify/app.sock;
        proxy_set_header    Host $host;
        proxy_set_header    X-Real-IP $remote_addr;
        proxy_read_timeout  65s;
        proxy_connect_timeout 5s;

    # ── JS/CSS/HTML/JSON — без кэша ───────────────────────────────
    location ~* \.(js|css|html|json)$ {

        add_header Pragma        "no-cache"                            always;
        add_header Expires       "0"                                   always;
        try_files $uri =404;
